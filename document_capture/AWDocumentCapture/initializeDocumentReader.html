To use the document capture service offline you will need to initialize the document reader with a license file. Returns true if the initilialization is complete

Implementation

Future initializeDocumentReader(String pathToLicense) async {
  ByteData? byteData = await rootBundle.load(pathToLicense);

  await DocumentReader.initializeReader({
    "license": base64.encode(byteData.buffer
        .asUint8List(byteData.offsetInBytes, byteData.lengthInBytes))
  }).then((s) {
    isInitilializedFunction;
    log(s.toString());
  }).catchError((Object error) {
    log((error as PlatformException).message.toString());
  });
  DocumentReader.setConfig({
    "customization": {"multipageButtonBackgroundColor": "#e02712"},
    "functionality": {
      "showSkipNextPageButton": showSkipNextPageButton,
    },
    "processParams": {
      "timeout": timeout,
      "timeoutFromFirstDetect": timeoutFromFirstDetect,
      "timeoutFromFirstDocType": timeoutFromFirstDocType,
      "scenario": scenarioIdentifier,
      "multipageProcessing": multiPageProcessing,
      "doublePageSpread": doublePageSpread,
      "checkHologram": checkHologram,
    },
  });
}
